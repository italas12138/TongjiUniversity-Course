int max(int x,int y){
    if(x>y){
        return x;
    }
    else{
        return y;
    }
}
int min(int x,int y){
    if(x<y){
        int t;
        t=x;
        x=y;
        y=t;
    }
    return y;
}
int sum(int x,int y){
    return x+y;
}
